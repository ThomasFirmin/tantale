pub mod sp_test {
    mod test {
        use serde::{Deserialize, Serialize};
        use tantale_core::domain::TypeDom;
        extern crate test;
        #[rustc_test_marker = "sp_test::test::main"]
        #[doc(hidden)]
        pub const main: test::TestDescAndFn = test::TestDescAndFn {
            desc: test::TestDesc {
                name: test::StaticTestName("sp_test::test::main"),
                ignore: false,
                ignore_message: ::core::option::Option::None,
                source_file: "tests_suite/tests/macros/sp_test.rs",
                start_line: 6usize,
                start_col: 12usize,
                end_line: 6usize,
                end_col: 16usize,
                compile_fail: false,
                no_run: false,
                should_panic: test::ShouldPanic::No,
                test_type: test::TestType::Unknown,
            },
            testfn: test::StaticTestFn(
                #[coverage(off)]
                || test::assert_test_result(main()),
            ),
        };
        pub fn main() {
            use std::sync::Arc;
            use tantale::core::{
                BasePartial, Bool, Cat, EmptyInfo, Nat, Real, SId, Searchspace, Solution,
                sampler::{Uniform, Bernoulli},
            };
            use tantale::macros::hpo;
            static ACTIVATION: [&str; 3] = ["relu", "tanh", "sigmoid"];
            use tantale::core::domain::{
                BaseDom, BaseTypeDom, Domain, NoDomain, onto::Onto,
            };
            pub type ObjType = BaseDom;
            pub type OptType = Real;
            pub fn get_searchspace() -> tantale::core::searchspace::Sp<BaseDom, Real> {
                let mut variables: Vec<
                    tantale::core::variable::var::Var<BaseDom, Real>,
                > = Vec::new();
                variables
                    .push({
                        tantale::core::variable::var::Var::<
                            BaseDom,
                            Real,
                        >::new(
                            ("a", None),
                            BaseDom::Real(Real::new(0.0, 1.0, Uniform)).into(),
                            Real::new(0.0, 1.0, Uniform).into(),
                        )
                    });
                variables
                    .push({
                        tantale::core::variable::var::Var::<
                            BaseDom,
                            Real,
                        >::new(
                            ("b", None),
                            BaseDom::Nat(Nat::new(0, 100, Uniform)).into(),
                            Real::new(0.0, 1.0, Uniform).into(),
                        )
                    });
                variables
                    .push({
                        tantale::core::variable::var::Var::<
                            BaseDom,
                            Real,
                        >::new(
                            ("c", None),
                            BaseDom::Cat(Cat::new(&ACTIVATION, Uniform)).into(),
                            Real::new(0.0, 1.0, Uniform).into(),
                        )
                    });
                variables
                    .push({
                        tantale::core::variable::var::Var::<
                            BaseDom,
                            Real,
                        >::new(
                            ("d", None),
                            BaseDom::Bool(Bool::new(Bernoulli(0.5))).into(),
                            Real::new(0.0, 1.0, Uniform).into(),
                        )
                    });
                tantale::core::searchspace::Sp {
                    var: variables.into_boxed_slice(),
                }
            }
        }
    }
}
